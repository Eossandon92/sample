samples('github:tidalcycles/dirt-samples')

const bass_pattern = "e2 f2 ~ g#1 ~ a1 f2 ~"

let main_bass = note(bass_pattern)
.sound("supersaw")
.sustain(0.3)
.release(0.1)
.shape(0.3)
.lpf(perlin.range(200, 1000).slow(4))
.lpq(slider(9.02, 3, 10))
.lpenv(slider(1, 1, 12))
.room(.1)
.postgain(1)

let sub_kick = s("bd:4!4").bank("tr808")
.shape(.6)
.sustain(.8)
.hpf(80)
.postgain(1)
let sub_kick2 = s("bd:4!4").bank("tr909")
.shape(.6)
.sustain(.8)
.hpf(80)
.postgain(1)

.lpf(742)
.shape(.2)
.sustain(.8)
.attack(0.005)
.hpf(80)
.postgain(1)

.sometimesBy(".3", x=>x.s("oh | tt"))
.velocity(perlin.range(.5, 5))
.gain(.3)

sub_kick,
main_kick,
hats,
cuenta_regresiva
